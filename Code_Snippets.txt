// Function to get an element by its inner text using XPath
function getElementByText(par_tag, tag, text) {
    var xpath = "//" + tag + "[text()='" + text + "']";
    var result = par_tag.evaluate(xpath, par_tag, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null);
    return result.singleNodeValue;
}

function typeText(textStr){
	var key = '';
	var code = '';
	var keyCode = 0;
	for(i=0;i<textStr.length;i++){
		key = textStr.substring(i,i+1);
	  	//document.getElementById("demo").innerHTML += "<br>"+key;
		if(parseFloat(key)!=NaN){
			code = 'Digit' + key;
			keyCode = 48 + parseFloat(key);
			// Create the KeyboardEvent
			var numberKeyEvent = new KeyboardEvent('keydown', {
				key: key,
				code: code,
				keyCode: keyCode,
				which: keyCode,
				bubbles: true,
				cancelable: true
			});	
		}
		document.dispatchEvent(numberKeyEvent);
	}
}
